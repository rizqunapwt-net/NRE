### 1.1 Automated Workflow Engine

#### **Book Publishing → Marketplace Auto-Population**

**Implementation Strategy:**

```php
// app/Observers/BookObserver.php
<?php

namespace App\Observers;

use App\Models\Book;
use App\Models\MarketplaceProduct;
use App\Services\InventoryService;
use App\Events\BookPublished;

class BookObserver
{
    public function __construct(
        private InventoryService $inventoryService
    ) {}

    public function updated(Book $book)
    {
        // Detect status change to 'Published'
        if ($book->isDirty('status') && $book->status === 'published') {
            $this->handleBookPublication($book);
        }
    }

    private function handleBookPublication(Book $book)
    {
        // 1. Auto-create Marketplace entry
        MarketplaceProduct::firstOrCreate(
            ['book_id' => $book->id],
            [
                'title' => $book->title,
                'price' => $book->retail_price,
                'isbn' => $book->isbn,
                'category_id' => $book->category_id,
                'status' => 'active',
                'stock_quantity' => $book->initial_stock ?? 0,
                'published_at' => now(),
            ]
        );

        // 2. Initialize Inventory
        $this->inventoryService->createInitialStock([
            'book_id' => $book->id,
            'quantity' => $book->initial_stock ?? 0,
            'location' => 'main_warehouse',
            'cost_per_unit' => $book->production_cost,
        ]);

        // 3. Dispatch Event for further automation
        event(new BookPublished($book));
        
        // 4. Log Activity
        activity()
            ->performedOn($book)
            ->causedBy(auth()->user())
            ->withProperties([
                'status_change' => 'published',
                'marketplace_created' => true,
                'inventory_initialized' => true,
            ])
            ->log('Book automatically published to marketplace');
    }
}
```

### 1.2 Financial Integration: Sales → Royalty Auto-Calculation

#### **Sales Import Triggers Automatic Royalty Calculation**

```php
// app/Services/RoyaltyCalculationService.php
<?php

namespace App\Services;

use App\Models\Sale;
use App\Models\Royalty;
use App\Models\Book;
use App\Events\RoyaltyCalculated;

class RoyaltyCalculationService
{
    public function processNewSale(Sale $sale): void
    {
        $book = Book::find($sale->book_id);
        
        if (!$book || !$book->author_id) {
            return;
        }

        $contract = $book->author->activeContract;
        
        if (!$contract) {
            return;
        }

        // Calculate royalty based on contract terms
        $royaltyAmount = $this->calculateRoyalty(
            $sale->net_price,
            $contract->royalty_percentage,
            $sale->quantity
        );

        // Create or update royalty record
        $royalty = Royalty::updateOrCreate(
            [
                'author_id' => $book->author_id,
                'book_id' => $book->id,
                'period' => now()->format('Y-m'),
            ],
            [
                'total_sales' => \DB::raw('total_sales + ' . $sale->quantity),
                'total_revenue' => \DB::raw('total_revenue + ' . $sale->total_amount),
                'royalty_amount' => \DB::raw('royalty_amount + ' . $royaltyAmount),
                'status' => 'pending',
            ]
        );

        event(new RoyaltyCalculated($royalty, $sale));
        
        // Auto-log for audit
        activity()
            ->performedOn($royalty)
            ->causedBy(auth()->user() ?? 'system')
            ->withProperties([
                'sale_id' => $sale->id,
                'royalty_amount' => $royaltyAmount,
                'calculation_method' => 'automatic',
            ])
            ->log('Royalty auto-calculated from sale');
    }

    private function calculateRoyalty(
        float $netPrice,
        float $royaltyPercentage,
        int $quantity
    ): float {
        return ($netPrice * $quantity) * ($royaltyPercentage / 100);
    }
}
```

#### **Middleware for Route Protection**

```php
// app/Http/Middleware/CheckSensitiveDataAccess.php
<?php

namespace App\Http\Middleware;

use Closure;
use Illuminate\Http\Request;

class CheckSensitiveDataAccess
{
    public function handle(Request $request, Closure $next, string $permission)
    {
        if (!auth()->user()->hasPermissionTo($permission)) {
            activity()
                ->causedBy(auth()->user())
                ->withProperties([
                    'attempted_permission' => $permission,
                    'ip' => $request->ip(),
                    'url' => $request->fullUrl(),
                ])
                ->log('Unauthorized access attempt to sensitive data');

            abort(403, 'Unauthorized access to sensitive information.');
        }

        return $next($request);
    }
}
```

#### **Critical Actions Audit**

```php
// app/Services/AuditService.php
<?php

namespace App\Services;

use Spatie\Activitylog\Models\Activity;

class AuditService
{
    public function logSensitiveAction(
        string $action,
        $subject,
        array $properties = []
    ): void {
        activity()
            ->causedBy(auth()->user())
            ->performedOn($subject)
            ->withProperties(array_merge($properties, [
                'ip_address' => request()->ip(),
                'user_agent' => request()->userAgent(),
                'timestamp' => now(),
                'severity' => 'high',
            ]))
            ->log($action);
    }

    public function getCriticalActivities($days = 30)
    {
        return Activity::where('created_at', '>=', now()->subDays($days))
            ->where('properties->severity', 'high')
            ->with('causer')
            ->orderBy('created_at', 'desc')
            ->get();
    }

    public function getActivityLog($subjectType, $subjectId)
    {
        return Activity::where('subject_type', $subjectType)
            ->where('subject_id', $subjectId)
            ->with('causer')
            ->orderBy('created_at', 'desc')
            ->get();
    }
}
```

```php
// app/Services/GlobalSearchService.php
<?php

namespace App\Services;

use App\Models\Book;
use App\Models\Author;
use App\Models\Employee;
use App\Models\Sale;
use Illuminate\Support\Collection;

class GlobalSearchService
{
    public function search(string $query, array $filters = []): array
    {
        $results = [];

        if (auth()->user()->can('view-books')) {
            $results['books'] = $this->searchBooks($query);
        }

        if (auth()->user()->can('view-authors')) {
            $results['authors'] = $this->searchAuthors($query);
        }

        if (auth()->user()->can('view-employees')) {
            $results['employees'] = $this->searchEmployees($query);
        }

        if (auth()->user()->can('view-sales')) {
            $results['sales'] = $this->searchSales($query);
        }

        return $results;
    }

    private function searchBooks(string $query): Collection
    {
        return Book::where('title', 'like', "%{$query}%")
            ->orWhere('isbn', 'like', "%{$query}%")
            ->limit(5)
            ->get(['id', 'title', 'isbn', 'status']);
    }

    private function searchAuthors(string $query): Collection
    {
        return Author::where('name', 'like', "%{$query}%")
            ->orWhere('email', 'like', "%{$query}%")
            ->limit(5)
            ->get(['id', 'name', 'email']);
    }

    private function searchEmployees(string $query): Collection
    {
        return Employee::where('name', 'like', "%{$query}%")
            ->orWhere('employee_id', 'like', "%{$query}%")
            ->limit(5)
            ->get(['id', 'name', 'employee_id', 'department_id']);
    }

    private function searchSales(string $query): Collection
    {
        return Sale::where('invoice_number', 'like', "%{$query}%")
            ->limit(5)
            ->get(['id', 'invoice_number', 'total_amount', 'date']);
    }
}
```

```php
// app/Console/Commands/CheckExpiringContracts.php
<?php

namespace App\Console\Commands;

use Illuminate\Console\Command;
use App\Models\Contract;
use App\Notifications\ContractExpiringNotification;

class CheckExpiringContracts extends Command
{
    protected $signature = 'contracts:check-expiring';
    protected $description = 'Check for contracts expiring within 30 days';

    public function handle()
    {
        $expiringContracts = Contract::where('end_date', '<=', now()->addDays(30))
            ->where('end_date', '>=', now())
            ->where('status', 'active')
            ->with('author')
            ->get();

        foreach ($expiringContracts as $contract) {
            $this->notifyRelevantUsers($contract);
        }

        $this->info("Checked {$expiringContracts->count()} expiring contracts.");
    }

    private function notifyRelevantUsers(Contract $contract)
    {
        // Notify Legal team
        $legalTeam = \App\Models\User::role('Legal')->get();
        
        foreach ($legalTeam as $user) {
            $user->notify(new ContractExpiringNotification($contract));
        }

        // Notify assigned Editor if exists
        if ($contract->author->editor_id) {
            $contract->author->editor->notify(new ContractExpiringNotification($contract));
        }
    }
}
```

```php
// app/Services/AnalyticsService.php
<?php

namespace App\Services;

use App\Models\Sale;
use App\Models\Employee;
use App\Models\Author;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Cache;

class AnalyticsService
{
    public function getMonthlyRevenueTrend(int $months = 12): array
    {
        return Cache::remember("revenue_trend_{$months}", 3600, function () use ($months) {
            $data = Sale::select(
                    DB::raw('DATE_FORMAT(date, "%Y-%m") as month'),
                    DB::raw('SUM(total_amount) as revenue'),
                    DB::raw('COUNT(*) as transaction_count')
                )
                ->where('date', '>=', now()->subMonths($months))
                ->groupBy('month')
                ->orderBy('month')
                ->get();

            return [
                'labels' => $data->pluck('month'),
                'revenue' => $data->pluck('revenue'),
                'transactions' => $data->pluck('transaction_count'),
                'forecast' => $this->generateForecast($data->pluck('revenue')->toArray()),
            ];
        });
    }

    private function generateForecast(array $historicalData): array
    {
        if (count($historicalData) < 3) {
            return ['next_month' => 0, 'confidence' => 'low'];
        }

        // Simple moving average forecast
        $lastThreeMonths = array_slice($historicalData, -3);
        $average = array_sum($lastThreeMonths) / count($lastThreeMonths);
        
        // Calculate trend
        $trend = ($lastThreeMonths[2] - $lastThreeMonths[0]) / 2;
        
        return [
            'next_month' => round($average + $trend, 2),
            'confidence' => 'moderate',
            'growth_rate' => round(($trend / $average) * 100, 2),
        ];
    }

    public function getDepartmentPerformance(): array
    {
        return Cache::remember('department_performance', 3600, function () {
            return DB::table('employees')
                ->join('attendances', 'employees.id', '=', 'attendances.employee_id')
                ->join('departments', 'employees.department_id', '=', 'departments.id')
                ->select(
                    'departments.name as department',
                    DB::raw('COUNT(DISTINCT employees.id) as employee_count'),
                    DB::raw('AVG(CASE WHEN attendances.status = "present" THEN 1 ELSE 0 END) * 100 as attendance_rate'),
                    DB::raw('SUM(CASE WHEN attendances.status = "late" THEN 1 ELSE 0 END) as late_count')
                )
                ->where('attendances.date', '>=', now()->subDays(30))
                ->groupBy('departments.id', 'departments.name')
                ->get()
                ->toArray();
        });
    }

    public function getAuthorRanking(): array
    {
        return Cache::remember('author_ranking', 3600, function () {
            return DB::table('authors')
                ->join('books', 'authors.id', '=', 'books.author_id')
                ->join('sales', 'books.id', '=', 'sales.book_id')
                ->select(
                    'authors.id',
                    'authors.name',
                    DB::raw('COUNT(DISTINCT books.id) as book_count'),
                    DB::raw('SUM(sales.total_amount) as total_revenue'),
                    DB::raw('SUM(sales.quantity) as total_copies_sold'),
                    DB::raw('AVG(sales.total_amount / sales.quantity) as avg_price_per_book')
                )
                ->groupBy('authors.id', 'authors.name')
                ->orderBy('total_revenue', 'desc')
                ->limit(20)
                ->get()
                ->toArray();
        });
    }

    public function getProductivityMetrics(): array
    {
        return [
            'books_published_this_month' => DB::table('books')
                ->where('status', 'published')
                ->whereMonth('created_at', now()->month)
                ->count(),
            
            'manuscripts_in_review' => DB::table('manuscripts')
                ->where('status', 'in_review')
                ->count(),
            
            'pending_royalties' => DB::table('royalties')
                ->where('status', 'pending')
                ->sum('royalty_amount'),
            
            'active_employees' => Employee::where('status', 'active')->count(),
        ];
    }

    public function getSalesByCategory(): array
    {
        return DB::table('sales')
            ->join('books', 'sales.book_id', '=', 'books.id')
            ->join('categories', 'books.category_id', '=', 'categories.id')
            ->select(
                'categories.name as category',
                DB::raw('SUM(sales.total_amount) as revenue'),
                DB::raw('SUM(sales.quantity) as units_sold')
            )
            ->where('sales.date', '>=', now()->subMonths(6))
            ->groupBy('categories.id', 'categories.name')
            ->orderBy('revenue', 'desc')
            ->get()
            ->toArray();
    }
}
```

```php
// app/Services/CacheService.php
<?php

namespace App\Services;

use Illuminate\Support\Facades\Cache;

class CacheService
{
    private const TTL = 3600; // 1 hour
    private const LONG_TTL = 86400; // 24 hours

    public function rememberDashboardData(string $userId, callable $callback)
    {
        return Cache::remember(
            "dashboard.{$userId}",
            self::TTL,
            $callback
        );
    }

    public function clearUserCache(string $userId): void
    {
        Cache::forget("dashboard.{$userId}");
        Cache::forget("revenue_trend_12");
        Cache::forget("department_performance");
        Cache::forget("author_ranking");
    }

    public function warmUpCache(): void
    {
        // Pre-compute expensive queries
        Cache::put('total_books_count', \App\Models\Book::count(), self::LONG_TTL);
        Cache::put('total_revenue_ytd', \App\Models\Sale::whereYear('date', now()->year)->sum('total_amount'), self::TTL);
        Cache::put('active_authors_count', \App\Models\Author::whereHas('contracts', function($q) {
            $q->where('status', 'active');
        })->count(), self::LONG_TTL);
    }

    public function clearAllCache(): void
    {
        Cache::flush();
    }
}
```

```jsx
                        {dept.employee_count} employees
                      </span>
                    </div>
                    <div className="flex items-center gap-2">
                      <div className="flex-1 h-2 bg-gray-200 rounded-full overflow-hidden">
                        <div 
                          className="h-full bg-green-500 rounded-full"
                          style={{ width: `${dept.attendance_rate}%` }}
                        />
                      </div>
                      <span className="text-sm font-semibold text-gray-700">
                        {dept.attendance_rate.toFixed(1)}%
                      </span>
                    </div>
                    {dept.late_count > 0 && (
                      <p className="text-xs text-orange-600">
                        {dept.late_count} late arrivals this month
                      </p>
                    )}
                  </div>
                ))}
              </div>
            </GlassCard>
          </div>

          {/* Sales by Category */}
          <GlassCard>
            <h3 className="text-xl font-bold text-gray-800 mb-6">
              Sales by Category (Last 6 Months)
            </h3>
            <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-5 gap-4">
              {sales_by_category.map((category, index) => (
                <div key={index} className="text-center p-4 bg-white/5 rounded-xl">
                  <p className="text-sm text-gray-600 mb-2">{category.category}</p>
                  <p className="text-2xl font-bold text-gray-800">
                    {category.units_sold}
                  </p>
                  <p className="text-xs text-gray-600 mt-1">
                    Rp {(category.revenue / 1000000).toFixed(1)}M
                  </p>
                </div>
              ))}
            </div>
          </GlassCard>
        </div>
      </div>
    </AuthenticatedLayout>
  );
}
```

// app/Services/Royalty/RoyaltyCalculatorService.php
namespace App\Services\Royalty;

use App\Models\Sale;
use App\Models\AuthorContract;
use App\Models\RoyaltyStatement;

class RoyaltyCalculatorService
{
    public function processImportedSale(Sale $sale)
    {
        $book = $sale->book;
        $contract = AuthorContract::where('author_id', $book->author_id)
            ->where('status', 'active')
            ->whereDate('end_date', '>=', now())
            ->first();
            
        if (!$contract) return;
        
        $royaltyRate = $this->determineRoyaltyRate($sale, $contract);
        $royaltyAmount = $this->calculateNetRoyalty($sale, $royaltyRate);
        
        // Auto-create or update royalty statement
        $statement = RoyaltyStatement::firstOrCreate(
            [
                'author_id' => $book->author_id,
                'period_month' => $sale->sale_date->format('Y-m'),
                'status' => 'draft'
            ],
            [
                'total_sales' => 0,
                'total_royalty' => 0,
                'currency' => $sale->currency
            ]
        );
        
        $statement->increment('total_sales', $sale->quantity);
        $statement->increment('total_royalty', $royaltyAmount);
        
        // Create detailed line item
        $statement->lineItems()->create([
            'sale_id' => $sale->id,
            'book_id' => $book->id,
            'quantity' => $sale->quantity,
            'net_price' => $sale->net_price,
            'royalty_rate' => $royaltyRate,
            'royalty_amount' => $royaltyAmount,
            'sale_date' => $sale->sale_date
        ]);
        
        // Trigger notification if threshold reached
        if ($statement->total_royalty >= $contract->minimum_payout_threshold) {
            event(new RoyaltyThresholdReached($statement));
        }
    }
    
    private function determineRoyaltyRate(Sale $sale, AuthorContract $contract)
    {
        // Tiered royalty logic based on format and volume
        $tier = $contract->tiers()
            ->where('format', $sale->format) // hardcover, paperback, ebook
            ->where('min_quantity', '<=', $sale->cumulative_quantity)
            ->where('max_quantity', '>=', $sale->cumulative_quantity)
            ->first();
            
        return $tier ? $tier->percentage : $contract->default_rate;
    }
}

namespace App\Services\Analytics;

use App\Models\Sale;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Collection;

class RevenueForecastService
{
    public function getMonthlyComparison(int $year = null): array
    {
        $year = $year ?? now()->year;
        
        $monthlyData = Sale::select(
            DB::raw('MONTH(sale_date) as month'),
            DB::raw('SUM(gross_revenue) as revenue'),
            DB::raw('SUM(net_revenue) as net_revenue'),
            DB::raw('COUNT(DISTINCT book_id) as unique_titles'),
            DB::raw('SUM(quantity) as units_sold')
        )
        ->whereYear('sale_date', $year)
        ->groupBy('month')
        ->orderBy('month')
        ->get()
        ->keyBy('month');
        
        // Calculate growth rates
        $previousYear = Sale::select(
            DB::raw('MONTH(sale_date) as month'),
            DB::raw('SUM(gross_revenue) as revenue')
        )
        ->whereYear('sale_date', $year - 1)
        ->groupBy('month')
        ->get()
        ->keyBy('month');
        
        $forecast = $this->calculateForecast($monthlyData);
        
        return [
            'current_year' => $this->formatMonthlyData($monthlyData),
            'previous_year' => $this->formatMonthlyData($previousYear),
            'growth_rate' => $this->calculateGrowthRates($monthlyData, $previousYear),
            'forecast_next_quarter' => $forecast,
            'trend' => $this->determineTrend($monthlyData)
        ];
    }
    
    protected function calculateForecast(Collection $data): array
    {
        // Simple linear regression for next 3 months
        $months = $data->keys()->toArray();
        $revenues = $data->pluck('revenue')->toArray();
        
        $n = count($months);
        $sumX = array_sum($months);
        $sumY = array_sum($revenues);
        $sumXY = 0;
        $sumX2 = 0;
        
        foreach ($months as $i => $x) {
            $sumXY += $x * $revenues[$i];
            $sumX2 += $x * $x;
        }
        
        $slope = ($n * $sumXY - $sumX * $sumY) / ($n * $sumX2 - $sumX * $sumX);
        $intercept = ($sumY - $slope * $sumX) / $n;
        
        $forecast = [];
        for ($i = 1; $i <= 3; $i++) {
            $nextMonth = max($months) + $i;
            $forecast[] = [
                'month' => $nextMonth,
                'predicted_revenue' => max(0, $slope * $nextMonth + $intercept),
                'confidence' => $this->calculateConfidence($data, $nextMonth)
            ];
        }
        
        return $forecast;
    }
}

import { createContext, useContext, useEffect, useState } from 'react'

interface GlassSettings {
  opacity: number
  blur: number
  saturation: number
  borderOpacity: number
  borderRadius: number
}

const defaultSettings: GlassSettings = {
  opacity: 0.08,
  blur: 16,
  saturation: 180,
  borderOpacity: 0.2,
  borderRadius: 24
}

const GlassContext = createContext<{
  settings: GlassSettings
  updateSettings: (s: Partial<GlassSettings>) => void
}>({ settings: defaultSettings, updateSettings: () => {} })

export function GlassProvider({ children }: { children: React.ReactNode }) {
  const [settings, setSettings] = useState<GlassSettings>(defaultSettings)
  
  useEffect(() => {
    const root = document.documentElement
    root.style.setProperty('--glass-opacity', settings.opacity.toString())
    root.style.setProperty('--glass-blur', `${settings.blur}px`)
    root.style.setProperty('--glass-saturation', `${settings.saturation}%`)
    root.style.setProperty('--glass-border', settings.borderOpacity.toString())
    root.style.setProperty('--glass-radius', `${settings.borderRadius}px`)
  }, [settings])
  
  return (
    <GlassContext.Provider value={{ 
      settings, 
      updateSettings: (s) => setSettings(prev => ({ ...prev, ...s })) 
    }}>
      {children}
    </GlassContext.Provider>
  )
}
